# formly custom template for a kendo date picker

# example usage for date picker with day, month and years:
{
    "key": "birthdate",
    "type": "kendo-date-picker",
    "templateOptions": {
        "label": "Birthdate:"
        "format": "dd/MMM/yyyy"
    }
}

# example usage for date picker with manth year only:
{
    "key": "gradutationDate",
    "type": "kendo-date-picker",
    "templateOptions": {
        "label": "Graduation Date:"
        "format": "MMM/yyyy"
    }
}
